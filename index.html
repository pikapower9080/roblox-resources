local frames = {"rbxassetid://9231227426","rbxassetid://9231227484","rbxassetid://9231227553","rbxassetid://9231227629","rbxassetid://9231227765","rbxassetid://9231227851","rbxassetid://9231227936","rbxassetid://9231228039","rbxassetid://9231228112","rbxassetid://9231228204"}

while true do
    for _,frame in pairs(frames) do
        script.Parent.Image = frame
        wait(0.04)
    end
end
